应该是力扣最经典的一道题目了，没什么好说的。
```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    const map = new Map();
    for(let i = 0; i < nums.length; i++) {
        const temp = map.get(target - nums[i]);
        if(temp !== undefined) {
            return [i, temp];
        } else {
            map.set(nums[i], i);
        }
    }
};
```
