第一种做法：反转以后正序遍历
这是我第一反应的做法，然后自然也想到了倒序遍历，不过没写出来，然后去看了一下题解
```js
/**
 * @param {string} num1
 * @param {string} num2
 * @return {string}
 */

// 模拟
var addStrings = function(num1, num2) {
    let carry = 0;
    const res = []
    const maxLength = Math.max(num1.length, num2.length);
    num1 = num1.split('').reverse().join('');
    num2 = num2.split('').reverse().join('');
    for(let i = 0; i < maxLength; i++) {
        let a = num1[i], b = num2[i];
        if(!a) {
            a = 0;
        }
        if(!b) {
            b = 0;
        }
        let sum = Number(a) + Number(b) + carry;
        carry = 0;
        while(sum >= 10) {
            carry++;
            sum -= 10;
        }
        res.push(sum);
    }
    if(carry) {
        res.push(carry);
    }
    return res.reverse().join('');
};
```

第二种做法：倒序遍历
```js
/**
 * @param {string} num1
 * @param {string} num2
 * @return {string}
 */

// 倒序遍历
var addStrings = function(num1, num2) {
    let i = num1.length - 1, j = num2.length - 1;
    const res = [];
    let carry = 0;
    while(i >= 0 || j >= 0 || carry > 0) {
        const a = i >= 0 ? num1.charAt(i) - '0' : 0;
        const b = j >= 0 ? num2.charAt(j) - '0' : 0;
        const sum = a + b + carry;
        res.push(sum % 10);
        carry = Math.floor(sum / 10);
        i--;
        j--;
    }
    return res.reverse().join('');
};
```
