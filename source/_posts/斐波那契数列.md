```js
/**
 * @param {number} n
 * @return {number}
 */
var fib = function(n) {
    if(n === 0) {
        return 0;
    }
    if(n === 1) {
        return 1;
    }
    
    let prev1 = 0, prev2 = 1, cur;
    for(let i = 2; i <= n; i++) {
        cur = ((prev1 % 1000000007) + (prev2 % 1000000007)) % 1000000007;
        prev1 = prev2;
        prev2 = cur;
    }
    
    return cur;
};
```
